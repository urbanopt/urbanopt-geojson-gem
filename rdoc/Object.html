def get_building_schema(strict)
  result = nil
  File.open(File.dirname(__FILE__) + '/../schema/building_properties.json') do |f|
    result = JSON.parse(f.read)
  end
  if strict
    result['additionalProperties'] = false
  else
    result['additionalProperties'] = true
  end
  return result
end
